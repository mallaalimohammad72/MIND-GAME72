let deferredPrompt; window.addEventListener("beforeinstallprompt", (event) => { event.preventDefault(); deferredPrompt = event; document.getElementById("installButton").style.display = "block"; }); function installApp() { if (deferredPrompt) { deferredPrompt.prompt(); deferredPrompt.userChoice.then((choiceResult) => { if (choiceResult.outcome === "accepted") { console.log("User installed the app"); } deferredPrompt = null; }); } }
